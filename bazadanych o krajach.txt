CREATE DATABASE swiat;
CREATE TABLE Kraje ( id int AUTO_INCREMENT PRIMARY KEY, nazwa varchar(50), ludnosc int(12), waluta varchar(20), proc_bezrobocia float, wielkosc_w_km² float(9), dlg_granicy_w_km int(5), prezydent varchar(80), PKB int(20) );
CREATE TABLE Kontynenty( id int AUTO_INCREMENT PRIMARY KEY, nazwa varchar(25), ludnosc int(12), najwiekszy_kraj varchar(50), wielkosc_w_km² float(12) );
CREATE TABLE miasta ( id int AUTO_INCREMENT PRIMARY KEY, nazwa varchar(50), ilość_miesz int(15), dlugosc_w_km float(10), kraj_w_ktorym_lezy varchar(20), rok_uzyskania_praw_miasta year(4) );
